#include <drivers\drv_gl.txt>

void Main()
{
    glVertexMode( 1 );
    glColor( 255, 255, 255, 255 );

    glBindTexture( 'brick/brick_model' );
    glColor( 255, 255, 255, 255 );
    glRectWH( 128, 128, 256, 256 );
    
    glExit(); 
}
