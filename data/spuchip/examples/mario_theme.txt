void main()
{
    // Tempo
    if ((i > 120) && (i <= 230))
        tempo( 1000 );
    else
        tempo( 864 );

    // Track A
    note = 2;
    fpwr note,(trackA[i]/12);
    note /= 100;
    chpitch 0,note;

    // Track B
    note = 2;
    fpwr note,(trackB[i]/12);
    note /= 100;
    chpitch 1,note;

    // Bass
    note = 2;
    fpwr note,(bass[i]/12);
    note /= 100;
    chpitch 2,note;

    // Index
    i++; mod i,tracklen;

    // Repeat
    jmp main;
}

// Accurate tempo function for beats-per-minute
void tempo( float bpm )
{
    timer timestamp;
    while ((time - timestamp) < (60 / bpm)) { timer time; }
}

// Returns the length of a string
float strlen(char* str)
{
  char* strptr = str;
  while (*strptr++);
  return (strptr - str);
}
